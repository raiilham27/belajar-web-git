// Window Screen // The window.screen object can be written without the window prefix. // Properties: // screen.width // screen.height // screen.availWidth // screen.availHeight // screen.colorDepth // screen.pixelDepth // Window Screen Width // The screen.width property returns the width of the visitor's screen in pixels. // Example // Display the width of the screen in pixels: // document.getElementById("demo").innerHTML = // "Screen Width: " + screen.width; // Result will be: // Screen Width: 1280 // Window Screen Height // The screen.height property returns the height of the visitor's screen in pixels. // Example // Display the height of the screen in pixels: // document.getElementById("demo").innerHTML = // "Screen Height: " + screen.height; // Result will be: // Screen Height: 720
