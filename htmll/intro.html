// avaScript Class Syntax // Use the keyword class to create a class. // Always add a method named constructor(): // Syntax // class ClassName { // constructor() { ... } // } // Example // class Car { // constructor(name, year) { // this.name = name; // this.year = year; // } // } // The example above creates a class named "Car". // The class has two initial properties: "name" and "year". // A JavaScript class is not an object. // It is a template for JavaScript objects. // Using a Class // When you have a class, you can use the class to create objects: // Example // const myCar1 = new Car("Ford", 2014); // const myCar2 = new Car("Audi", 2019);
